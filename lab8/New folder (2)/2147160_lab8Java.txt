import java.util.Scanner;

//Multiple Inheritance
interface Aircraft {
    void aircrafttype();
}

interface Airline {
    void fare();
}

class Flight implements Aircraft, Airline {
    public void aircrafttype() {
        System.out.println("This is a boeing Airline");
    }

    public void fare() {
        System.out.println("The fare for boeing airline is usually more than the airbus");
    }

}

// Extending Interface
interface Cargo {
    void bookItems(int x);

    // static message inside interface
    static void printMessage() {
        System.out.println("Welcome to the Cargo Service");
    }
}

interface CargoAirline extends Cargo {
    void airlinename();
}

class bookItem implements CargoAirline {
    public void bookItems(int x) {
        if (x > 10) {
            System.out.println("Price will be 1000 Rs");
        } else if (x < 10) {
            System.out.println("Price will be 499");
        } else {
            System.out.println("Can't get exact weight,please try again!");
        }
    }

    public void airlinename() {
        System.out.println("This is lufthansa cargo\n");
    }
}

// nested interface
interface ticket {
    void bookTicket(int x);

    interface payment {
        void message();
    }
}

class transaction implements ticket.payment {
    Scanner sc = new Scanner(System.in);

    public void bookTicket(int x) {
        System.out.println("Indigo is requesting for a payment of Rs \n" + x);
        System.out.println("\nPress 1 to continue\n");
        int choice = sc.nextInt();
        if (choice == 1) {
            System.out.println("Do not refresh this page......\n");
        } else {
            System.exit(0);
        }

    }

    public void message() {
        System.out.println("Your payment was successful,thank you for choosing us.We look to serve you!\n");
    }

}
